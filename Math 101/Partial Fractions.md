$$
\int 1-u^{2} \, du
$$
$$
= \int \frac{1}{(1-u)(1+u)} \, du
$$
$$
\frac{1}{(1-u)(1+u)} = \frac{A}{1-u}+\frac{B}{1+u} \implies 1 = A(1+u) + B(1-u)
$$
$$
\int \frac{1}{(1-u^{2})} \, du  = \frac{1}{2}\ln|(\frac{1+u}{1-u})|
$$
